## [1137.第-n-个泰波那契数](https://leetcode.cn/problems/n-th-tribonacci-number/) <Badge type="error">hard</Badge>


```js
/**
 * @param {number} n
 * @return {number}
 */
var tribonacci = function (n) {
  const dp = new Array(n + 1).fill(0);
  dp[0] = 0;
  dp[1] = 1;
  dp[2] = 1;
  for (let i = 3; i <= n; i++) {
    dp[i] = dp[i - 1] + dp[i - 2] + dp[i - 3];
  }
  return dp[n];
};
```
